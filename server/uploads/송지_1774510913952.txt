CREATE TABLE `Survey_main` (
   `main_no`   int   NOT NULL,
   `version_no`   int   NOT NULL,
   `main_title`   varchar   NOT NULL
);

CREATE TABLE `survey_version` (
   `version_no`   int   NOT NULL,
   `version_name`   varchar   NOT NULL,
   `is_active`   boolean   NOT NULL,
   `created_at`   datetime   NOT NULL   DEFAULT sysdate,
   `finish_date`   datetime   NULL
);

CREATE TABLE `beneficiaries` (
   `beneficiaries_no`   int   NOT NULL,
   `guardian_no`   int   NOT NULL,
   `manager_no`   int   NULL,
   `sub_manager_no`   int   NOT NULL,
   `institution_no`   int   NOT NULL,
   `beneficiaries_name`   varchar   NOT NULL,
   `birth`   date   NOT NULL,
   `tel`   varchar   NULL,
   `gender`   varchar   NOT NULL,
   `beneficiaries_address`   varchar   NOT NULL,
   `disability_type`   varchar   NOT NULL,
   `relation`   varchar   NOT NULL,
   `created_at`   datetime   NOT NULL   DEFAULT sysdate
);

CREATE TABLE `institution` (
   `institution_no`   int   NOT NULL,
   `name`   varchar   NOT NULL,
   `institution_tel`   varchar   NOT NULL,
   `institution_address`   varchar   NOT NULL,
   `institution_email`   varchar   NOT NULL,
   `tel`   varchar   NOT NULL,
   `operation`   boolean   NOT NULL,
   `created_at`   datetime   NULL   DEFAULT sysdate
);

CREATE TABLE `result_history` (
   `result_history_no`   int   NOT NULL,
   `support_result_no`   int   NULL,
   `history_title`   varchar   NOT NULL,
   `history_content`   varchar   NOT NULL,
   `history_writer`   varchar   NOT NULL,
   `approval`   varchar   NOT NULL,
   `created_at`   datetime   NOT NULL   DEFAULT sysdate,
   `finish`   varchar   NOT NULL,
   `support_result_no2`   int   NOT NULL
);

CREATE TABLE `Survey_sub` (
   `sub_no`   int   NOT NULL,
   `main_no`   int   NOT NULL,
   `sub_title`   varchar   NOT NULL,
   `describe`   varchar   NULL
);
